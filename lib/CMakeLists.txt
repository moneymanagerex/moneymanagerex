IF(NOT InMMEX)
    MESSAGE(FATAL_ERROR "Use the top-level CMake script!")
ENDIF(NOT InMMEX)

FILE(GLOB LUA_SRC
        "${CMAKE_CURRENT_SOURCE_DIR}/lua/src/*.c"
        )

TARGET_SOURCES(LUA PRIVATE ${LUA_SRC})

TARGET_SOURCES(wxSQLite3 PRIVATE
        "${CMAKE_CURRENT_SOURCE_DIR}/wxsqlite3/src/wxsqlite3.cpp"
        "${CMAKE_CURRENT_SOURCE_DIR}/wxsqlite3/sqlite3/secure/src/sqlite3secure.c"
        )

TARGET_COMPILE_DEFINITIONS(wxSQLite3 PRIVATE
        NOPCH
        SQLITE_CORE
        SQLITE_ENABLE_FTS3
        SQLITE_ENABLE_FTS3_PARENTHESIS
        SQLITE_HAS_CODEC
        SQLITE_ENABLE_EXTFUNC
        SQLITE_ENABLE_COLUMN_METADATA
        SQLITE_ENABLE_JSON1
        HAVE_ACOSH
        HAVE_ASINH
        HAVE_ATANH
        HAVE_ISBLANK
        WXSQLITE3_HAVE_CODEC
        WXSQLITE3_HAVE_METADATA
        WXSQLITE3_USER_AUTHENTICATION
        )

IF(NOT WIN32)
        TARGET_COMPILE_DEFINITIONS(LUA PRIVATE
                LUA_USE_POSIX
                LUA_USE_DLOPEN
                LUA_USE_STRTODHEX
                LUA_USE_AFORMAT
                LUA_USE_LONGLONG
                )
ENDIF()
